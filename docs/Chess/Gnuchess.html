def gnuchess_move
  pipe = IO.popen('gnuchess -x', 'r+')
  begin
    pipe.puts('depth 1')
    pipe.puts('manual')
    self.coord_moves.each do |m|
      pipe.puts(m)
    end
    pipe.puts('go')
    while (line = pipe.gets)
      raise IllegalMoveError if line.include?('Invalid move')

      match = line.match(/My move is : ([a-h][1-8][a-h][1-8][rkbq]?)/)
      return match[1] if match
    end
  ensure
    pipe.puts('quit')
    pipe.close
  end
  return moves
end
